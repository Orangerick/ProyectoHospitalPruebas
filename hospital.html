📂 Esquema Relacional de Persistencia Clínica

Diseño del modelo de datos del sistema. Aunque la implementación usa almacenamiento en JSON, las relaciones siguen el modelo relacional clásico.

-- Tabla principal de usuarios del sistema (todos los roles) USUARIOS ( id VARCHAR(50) PRIMARY KEY, -- en este sistema: user nombre VARCHAR(100), email VARCHAR(100), rol ENUM('admin','doctor','paciente'), activo BOOLEAN, pass VARCHAR(64) NOT NULL, -- hash + salt telefono VARCHAR(20), direccion VARCHAR(200), foto TEXT, -- base64 creado TIMESTAMP ) -- Catálogo de especialidades médicas ESPECIALIDADES ( id INT PRIMARY KEY, nombre_especialidad VARCHAR(100) NOT NULL UNIQUE ) -- Un médico ES UN usuario (relación 1:1 con USUARIOS) MEDICOS ( usuario_id VARCHAR(50) PRIMARY KEY, FOREIGN KEY → USUARIOS(id), cedula_professional VARCHAR(50) UNIQUE NOT NULL, especialidad_id INT FOREIGN KEY → ESPECIALIDADES(id) ) PACIENTES ( id BIGINT PRIMARY KEY, usuario_id VARCHAR(50) FOREIGN KEY → USUARIOS(id) -- opcional, si tiene login, nombre VARCHAR(100), edad INT, genero ENUM('M','F','O'), diagnostico VARCHAR(200), telefono VARCHAR(20), alergias VARCHAR(200) ) CITAS ( id BIGINT PRIMARY KEY, pid BIGINT FOREIGN KEY → PACIENTES(id) ON DELETE CASCADE, mid VARCHAR(50) FOREIGN KEY → MEDICOS(usuario_id) ON DELETE SET NULL, fecha DATE, hora TIME, motivo VARCHAR(200), estado ENUM('Pendiente','Atendida','Cancelada') ) HISTORIAL_CLINICO ( id BIGINT PRIMARY KEY, pid BIGINT FOREIGN KEY → PACIENTES(id) ON DELETE CASCADE, mid VARCHAR(50) FOREIGN KEY → MEDICOS(usuario_id) ON DELETE SET NULL, cita_id BIGINT FOREIGN KEY → CITAS(id) ON DELETE SET NULL, tipo ENUM('Consulta','Diagnóstico','Tratamiento','Estudio','Receta'), titulo VARCHAR(150), detalle TEXT, autor VARCHAR(50) FOREIGN KEY → USUARIOS(id), fecha TIMESTAMP ) LOG_AUDITORIA ( id BIGINT PRIMARY KEY, fecha TIMESTAMP, usuario VARCHAR(50) FOREIGN KEY → USUARIOS(id), accion VARCHAR(100), detalle VARCHAR(255) ) SESIONES ( token VARCHAR(64) PRIMARY KEY, user VARCHAR(50) FOREIGN KEY → USUARIOS(id), rol VARCHAR(20), inicio TIMESTAMP, expira TIMESTAMP )
